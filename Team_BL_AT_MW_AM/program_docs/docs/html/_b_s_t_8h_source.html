188 template <class T, class N>
189 BSTNode<T, N>* BST<T, N>::addHelper(BSTNode<T, N> *currentNode, N* val, std::function<T(N*)>* access, unsigned int &operations)
190 {
191  operations += 5;
192  //***PRINTING OUT
193  if (currentNode == nullptr)
194  {
195  currentNode = new BSTNode<T, N>;
196  currentNode->setValue(val);
197  currentNode->setHeight(0);
198  currentNode->setLeft(nullptr);
199  currentNode->setRight(nullptr);
200  }
201 
202  else if ((*access)(currentNode->getValue()) <= (*access)(val))
203  {
204  // val greater or equal than root to the left
205  //* if (!currentNode->getLeft ())
206  {
207  //currentNode->setLeft (new BSTNode<T, N> (val));
208  currentNode->setRight(addHelper(currentNode->getRight(), val, access, operations));
209  //******************************************************
210  // AVL TREE (By Ahn)
211  //******************************************************
212  if (heightHelper(currentNode->getRight()) - heightHelper(currentNode->getLeft()) == 2)
213  {
214  if ((*access)(val) >= (*access)(currentNode->getRight()->getValue()))
215  currentNode = singleLeftRotate(currentNode);
216  else
217  currentNode = doubleLeftRotate(currentNode);
218  }
219  }
220  //else addHelper (currentNode->getLeft (), val, access);
221  }
222  else
223  {
224  // val smaller to root to the right
225  //**if (!currentNode->getRight ())
226  {
227  //currentNode->setRight (new BSTNode<T, N> (val));
228  currentNode->setLeft(addHelper(currentNode->getLeft(), val, access, operations));
229  //******************************************************
230  // AVL TREE (By Ahn)
231  //******************************************************
232  if (heightHelper(currentNode->getLeft()) - heightHelper(currentNode->getRight()) == 2)
233  {
234  if ((*access)(val) < (*access)(currentNode->getLeft()->getValue()))
235  currentNode = singleRightRotate(currentNode);
236  else
237  currentNode = doubleRightRotate(currentNode);
238  }
239  }
240  //else addHelper (currentNode->getRight (), val, access);
241  }
242  setHeight(MathHelper::max(heightHelper(currentNode->getLeft()), heightHelper(currentNode->getRight())) + 1);
243  return currentNode;
244 }

346 unsigned int BST<T, N>::widthHelper(List<BSTNode<T, N>*>** levelNodePtrArr, unsigned int levelMax)
347 {
348  unsigned int i, j, n, n1, width;
349  width = 0;
350  for (i = 0; i < levelMax; i++)
351  {
352  n1 = 0;
353  n = levelNodePtrArr[i]->size();
354  for (j = 0; j < n; j++)
355  {
356  // only count non-null elements in list
357  if ((*levelNodePtrArr[i])[j])
358  n1++;
359  }
360  width = n1 > width ? n1 : width;
361  }
362  return width;
363 }

390 template <class T, class N>
391 bool BST<T, N>::removeHelper(BSTNode<T, N>* parent, BSTNode<T, N>* current, T value, std::function<T(N*)>* access)
392 {
393  if (!current) return false;
394  T tmp;
395  tmp = (*access)(current->getValue());
396  if (tmp == value)
397  {
398  // value found
399  return deleteHelper(parent, current)
400  }
401  else if (tmp < value) {
402  return removeHelper(current, current->getRight(), value, access);
403  }
404  else {
405  return removeHelper(current, current->getLeft(), value, access);
406  }
407 }

423 template <class T, class N>
424 bool BST<T, N>::deleteHelper(BSTNode<T, N>* parent, BSTNode<T, N>* current) {
425  if (current->getLeft() == NULL || current->getRight() == NULL)
426  {
427  // one child
428  BSTNode<T, N>* temp = current->getLeft();
429  if (current->getRight()) temp = current->getRight();
430  if (parent)
431  {
432  if (parent->getLeft() == current)
433  {
434  parent->setLeft(temp);
435  }
436  else
437  {
438  parent->setRight(temp);
439  }
440  }
441  else
442  {
443  this->root = temp;
444  }
445  }
446  else
447  {
448  // two children
449  BSTNode<T, N>* validSubs = current->getRight();
450  while (validSubs->getLeft())
451  {
452  validSubs = validSubs->getLeft();
453  }
454  N* temp = current->getValue();
455  current->setValue(validSubs->getValue());
456  validSubs->setValue(temp);
457  return removeHelper(current, current->getRight(), temp);
458  }
459  delete current;
460  return true;
461 }

463 template <class T, class N>
464 void BST<T, N>::findHelper(BSTNode<T, N>* current, T value, List<N*>* listPtr, std::function<T(N*)>* access, unsigned int &operations)
465 {
466  /*
467  searches the normal add path to find matching values
468  list will be empty if no matches found assuming list was
469  empty in the beginning
470  */
471  T tmp;
472  if (!current) return;
473  /* sometimes the access function may try to access an index or value out of range
474  the BST was probably constructed wrong or a value changed
475  */
476  try
477  {
478  tmp = (*access)(current->getValue());
479  }
480  catch (const std::exception& e)
481  {
482  throw std::runtime_error(e.what());
483  }
484  if (tmp == value)
485  listPtr->push_back(current->getValue());
486  if (tmp > value)
487  {
488  // val less than root to the left
489  operations++;
490  findHelper(current->getLeft(), value, listPtr, access, operations);
491  }
492  else
493  {
494  // val greater or equal to root to the right
495  operations++;
496  findHelper(current->getRight(), value, listPtr, access, operations);
497  }
498 }

615 template <class T, class N>
616 BSTNode<T, N>* BST<T, N>::removeAVLHelper(N* x, BSTNode<T, N>* t, std::function<T(N*)>* access)
617 {
618  BSTNode<T, N>* temp;
619 
620  // Element not found
621  if (t == nullptr)
622  return nullptr;
623 
624  // Searching for element
625  else if ((*access)(x) < (*access)t->getValue())
626  t->setLeft(removeAVLHelper(x, t->getLeft(), access));
627  else if (x > t->getValue())
628  t->setRight(removeAVLHelper(x, t->getRight(), access));
629 
630  // Element found
631  // With 2 children
632  else if (t->getLeft() && t->getRight())
633  {
634  temp = findMin(t->getRight());
635  t->setValue(temp->getValue());
636  t->getRight(removeAVLHelper(t->getValue(), t->getRight(), access));
637  }
638  // With one or zero child
639  else
640  {
641  temp = t;
642  if (t->getLeft() == nullptr)
643  t = t->getRight();
644  else if (t->getRight() == nullptr)
645  t = t->getLeft();
646  delete temp;
647  }
648 
649  if (t == nullptr)
650  return t;
651 
652  setHeight(MathHelper::max(heightHelper(t->getLeft()), heightHelper(t->getRight()) + 1));
653 
654  // If node is unbalanced
655  // If left node is deleted, right case
656  if (heightHelper(t->getLeft()) - heightHelper(t->getRight()) == 2)
657  {
658  // right right case
659  if (heightHelper(t->getLeft()->getLeft()) - heightHelper(t->getLeft()->getRight()) == 1)
660  return singleLeftRotate(t);
661  // right left case
662  else
663  return doubleLeftRotate(t);
664  }
665  // If right node is deleted, left case
666  else (heightHelper(t->getRight()) - heightHelper(t->getLeft()) == 2)
667  {
668  // left left case
669  if (heightHelper(t->getRight()->getRight()) - heightHelper(t->getRight()->getLeft()) == 1)
670  return singleRightRotate(t);
671  // left right case
672  else
673  return doubleRightRotate(t);
674  }
675  return t;
676 }
